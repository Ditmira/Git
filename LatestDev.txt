CREATE TABLE STDDIT_MAIN(
CUSTOMER_NUMBER NUMBER(20),
CUSTOMER_NAME VARCHAR2(10),
PHONE_NUMBER NUMBER(15),
EMAIL VARCHAR2(30),
CUSTOMER_ACCOUNT VARCHAR2(20),
CUSTOMER_PRIORITY VARCHAR2(10),
AMOUNT NUMBER(10),
CURRENCY VARCHAR2(10),
REFERENCE NUMBER(10),
RAPORT_DATE DATE,
CONSTRAINT PK_STDDIT_MASTER PRIMARY KEY (CUSTOMER_NUMBER, CUSTOMER_NAME, REFERENCE)
);

CREATE TABLE STDDIT_DETAIL(
SUBJECT VARCHAR2(20),
DESCRIPTION VARCHAR2(200),
USER_ID VARCHAR2(15),
DESCRIPTION_DATE DATE,
REFERENCE NUMBER(10),
CONSTRAINT PK_STDDIT_DETAIL PRIMARY KEY (USER_ID, REFERENCE)
);

CREATE TABLE STDDIT_HEADER(
REFERENCE NUMBER(10) PRIMARY KEY NOT NULL,
RAPORT_DATE DATE,
AUTH_STAT  CHAR(1) DEFAULT 'U',
RECORD_STAT  CHAR (1) DEFAULT 'O',
MOD_NO  NUMBER (4,0),
CHECKER_DT_STAMP  DATE,
ONCE_AUTH  CHAR(1) DEFAULT 'N',
MAKER_ID  VARCHAR2(12),
MAKER_DT_STAMP DATE,
CHECKER_ID VARCHAR2(12),
DESCRIPTION VARCHAR2(50)
);
